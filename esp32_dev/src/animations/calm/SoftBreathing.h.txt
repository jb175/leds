#ifndef SOFT_BREATHING_H
#define SOFT_BREATHING_H

#include "calmAnimation.h"
#include "leds.h"

class SoftBreathing : public CalmAnimation {
public:
    // Constructor with customizable breathing color and speed
    SoftBreathing(CRGB color = CRGB::Blue, unsigned int breathingSpeed = 12);
    ~SoftBreathing() override = default;

    // Generates the LED states for this animation
    void generateLEDs(CRGB leds[NUM_LEDS], unsigned long timeElapsed) override;

private:
    CRGB color;         // Color of the breathing effect
    unsigned int breathingSpeed; // Speed of breathing (lower = faster)
};

#endif // SOFT_BREATHING_H
