#include "SoftBreathing.h"
#include <FastLED.h>

// Constructor: Set the color and breathing speed
SoftBreathing::SoftBreathing(CRGB color, unsigned int breathingSpeed) 
    : color(color), breathingSpeed(breathingSpeed) {}

// Generate LED states for the SoftBreathing animation
void SoftBreathing::generateLEDs(CRGB leds[NUM_LEDS], unsigned long timeElapsed) {
    // Create a breathing effect by varying the brightness over time
    uint8_t brightness = (sin8(timeElapsed / breathingSpeed) / 2) + 128; // Breathing effect with sine wave

    // Apply the color and breathing brightness to each LED
    for (int i = 0; i < NUM_LEDS; i++) {
        leds[i] = color;
        leds[i].nscale8(brightness); // Adjust brightness for breathing effect
    }
}
